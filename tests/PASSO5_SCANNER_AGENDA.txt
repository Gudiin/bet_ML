def get_scheduled_matches(self, date_str: str = None, leagues: list = None) -> list:
    """
    Busca jogos agendados para uma data especÃ­fica.
    
    Args:
        date_str: Data no formato 'YYYY-MM-DD' (ex: '2025-12-02')
                  Se None, usa a data de hoje
        leagues: Lista de IDs de ligas para filtrar (opcional)
                 Ex: [325] para BrasileirÃ£o SÃ©rie A
    
    Returns:
        list: Lista de dicionÃ¡rios com informaÃ§Ãµes dos jogos:
            {
                'match_id': '13472604',

                'away_team': 'Fluminense',
                'tournament': 'BrasileirÃ£o SÃ©rie A',
                'start_time': '2025-12-02T20:00:00',
                'status': 'notstarted'
            }
    
    LÃ³gica:
        1. Converte date_str para timestamp Unix
        2. Faz request para API de eventos por data
        3. Filtra por ligas (se especificado)
        4. Retorna lista de jogos
    
    API Endpoint:
        https://www.sofascore.com/api/v1/sport/football/scheduled-events/{date}
        Onde {date} Ã© no formato 'YYYY-MM-DD'
    """
    from datetime import datetime, timedelta
    
    # Se nÃ£o especificou data, usa hoje
    if date_str is None:
        date_str = datetime.now().strftime('%Y-%m-%d')
    
    print(f"ğŸ” Buscando jogos agendados para: {date_str}")
    
    # Endpoint da API do SofaScore para jogos por data
    api_url = f"https://www.sofascore.com/api/v1/sport/football/scheduled-events/{date_str}"
    
    try:
        data = self._fetch_api(api_url)
        
        if not data or 'events' not in data:
            print("âŒ Nenhum jogo encontrado para esta data.")
            return []
        
        matches = []
        for event in data['events']:
            # Filtra por ligas se especificado
            if leagues and event.get('tournament', {}).get('uniqueTournament', {}).get('id') not in leagues:
                continue
            
            match_info = {
                'match_id': str(event['id']),
                'home_team': event['homeTeam']['name'],
                'away_team': event['awayTeam']['name'],
                'tournament': event.get('tournament', {}).get('name', 'Unknown'),
                'start_time': datetime.fromtimestamp(event['startTimestamp']).strftime('%Y-%m-%d %H:%M'),
                'status': event.get('status', {}).get('type', 'unknown')
            }
            matches.append(match_info)
        
        print(f"âœ… Encontrados {len(matches)} jogos para {date_str}")
        return matches
        
    except Exception as e:
        print(f"âŒ Erro ao buscar jogos: {e}")
        return []
